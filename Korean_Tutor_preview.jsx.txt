// Numbers drill sub-component — three modes
function NumbersDrill() {
  const [mode, setMode] = useState(null); // null | 'speed' | 'fill' | 'context'

  // Speed round state
  const [speedItems, setSpeedItems] = useState([]);
  const [speedIdx, setSpeedIdx] = useState(0);
  const [speedRevealed, setSpeedRevealed] = useState(false);
  const [speedScore, setSpeedScore] = useState({right:0,wrong:0});
  const [speedDone, setSpeedDone] = useState(false);
  const [speedTimer, setSpeedTimer] = useState(4);
  const timerRef = useRef(null);

  // Fill-in-the-blank state
  const [fillIdx, setFillIdx] = useState(0);
  const [fillAnswers, setFillAnswers] = useState([]);
  const [fillChecked, setFillChecked] = useState(false);

  // Context-first state
  const [ctxItems, setCtxItems] = useState([]);
  const [ctxIdx, setCtxIdx] = useState(0);
  const [ctxChosen, setCtxChosen] = useState(null);
  const [ctxScore, setCtxScore] = useState({right:0,wrong:0});
  const [ctxDone, setCtxDone] = useState(false);

  const shuf = (a) => { const b=[...a]; for(let i=b.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[b[i],b[j]]=[b[j],b[i]];} return b; };

  // Speed round timer
  const startSpeed = () => {
    clearTimeout(timerRef.current);
    const items = shuf([...SPEED_ITEMS]);
    setSpeedItems(items); setSpeedIdx(0); setSpeedRevealed(false);
    setSpeedScore({right:0,wrong:0}); setSpeedDone(false); setSpeedTimer(4);
    setMode('speed');
    setTimeout(runTimer, 50); // slight delay so state settles
  };

  const advanceSpeed = (correct) => {
    clearTimeout(timerRef.current);
    setSpeedScore(s=>({...s, right: s.right+(correct?1:0), wrong: s.wrong+(correct?0:1)}));
    const next = speedIdx + 1;
    if (next >= speedItems.length) { setSpeedDone(true); return; }
    setSpeedIdx(next); setSpeedRevealed(false); setSpeedTimer(4);
    setTimeout(runTimer, 50);
  };

  // Speed timer: decrement every second while in speed mode and not revealed
  // Use a simple recursive setTimeout stored in ref
  const runTimer = () => {
    clearTimeout(timerRef.current);
    timerRef.current = setTimeout(() => {
      setSpeedTimer(t => {
        if (t <= 1) { setSpeedRevealed(true); return 0; }
        runTimer();
        return t - 1;
      });
    }, 1000);
  };

  // Fill blank helpers
  const currentFill = FILL_BLANKS[fillIdx];
  const startFill = () => { setFillIdx(0); setFillAnswers([]); setFillChecked(false); setMode('fill'); };
  const handleFillPick = (blankIdx, val) => {
    if (fillChecked) return;
    const next = [...fillAnswers];
    next[blankIdx] = val;
    setFillAnswers(next);
  };
  const allFillAnswered = currentFill && fillAnswers.filter(a=>a!==undefined).length >= currentFill.blanks.length;
  const fillCorrect = currentFill && currentFill.blanks.every((b,i)=>fillAnswers[i]===b.answer);

  // Context-first helpers
  const startCtx = () => {
    setCtxItems(shuf([...CONTEXT_CARDS])); setCtxIdx(0); setCtxChosen(null);
    setCtxScore({right:0,wrong:0}); setCtxDone(false); setMode('context');
  };
  const currentCtx = ctxItems[ctxIdx];

  const systemColor = s => s==="native"?C.blueBright:C.red;
  const systemLabel = s => s==="native"?"Native 🔵":"Sino 🔴";

  if (!mode) return (
    <div style={{ padding:"8px 0" }}>
      <div style={{ fontSize:13, color:C.muted, marginBottom:16, lineHeight:1.6 }}>
        Three drills to master Korean numbers — both Native (하나·둘…) and Sino-Korean (일·이·삼…) in context.
        <div style={{ marginTop:8, display:"flex", gap:8, flexWrap:"wrap" }}>
          <span style={{ background:C.blueLight, color:C.blueBright, borderRadius:6, padding:"2px 8px", fontSize:11, fontWeight:700 }}>🔵 Native = hours, age, counters</span>
          <span style={{ background:"#FEF2F2", color:"#B83232", borderRadius:6, padding:"2px 8px", fontSize:11, fontWeight:700 }}>🔴 Sino = minutes, months, dates, money</span>
        </div>
      </div>
      <div style={{ display:"flex", flexDirection:"column", gap:10 }}>
        <div onClick={startCtx} style={{ background:C.card, border:`1.5px solid ${C.border}`, borderRadius:14, padding:"18px 16px", cursor:"pointer" }}>
          <div style={{ fontWeight:700, fontSize:15, color:C.blueBright, marginBottom:4 }}>🎯 Context-First Cards</div>
          <div style={{ fontSize:13, color:C.muted }}>See "3 o'clock" or "15th (date)" → choose the right Korean. Builds the system-switching reflex.</div>
          <div style={{ fontSize:11, color:C.muted, marginTop:6 }}>{CONTEXT_CARDS.length} cards · shuffled each round</div>
        </div>
        <div onClick={startFill} style={{ background:C.card, border:`1.5px solid ${C.border}`, borderRadius:14, padding:"18px 16px", cursor:"pointer" }}>
          <div style={{ fontWeight:700, fontSize:15, color:C.blueBright, marginBottom:4 }}>✏️ Fill in the Blank</div>
          <div style={{ fontSize:13, color:C.muted }}>Complete real Korean sentences by picking the correct number form. Hours and minutes in the same sentence.</div>
          <div style={{ fontSize:11, color:C.muted, marginTop:6 }}>{FILL_BLANKS.length} sentences · tap to choose each blank</div>
        </div>
        <div onClick={startSpeed} style={{ background:C.card, border:`1.5px solid ${C.border}`, borderRadius:14, padding:"18px 16px", cursor:"pointer" }}>
          <div style={{ fontWeight:700, fontSize:15, color:C.blueBright, marginBottom:4 }}>⚡ Speed Round</div>
          <div style={{ fontSize:13, color:C.muted }}>A number flashes on screen — read it in Korean before the timer runs out. Tap ✓ or ✕ to track yourself.</div>
          <div style={{ fontSize:11, color:C.muted, marginTop:6 }}>{SPEED_ITEMS.length} items · 4 seconds each</div>
        </div>
      </div>
    </div>
  );

  // ── CONTEXT-FIRST MODE ────────────────────────────────────────────────────
  if (mode === 'context') {
    if (ctxDone) return (
      <div style={{ textAlign:"center", padding:24 }}>
        <div style={{ fontSize:48 }}>{ctxScore.right>=Math.round(ctxItems.length*0.85)?"🏆":"⭐"}</div>
        <div style={{ fontSize:28, fontWeight:700, color:C.blueBright, marginTop:8 }}>{ctxScore.right}/{ctxItems.length}</div>
        <div style={{ color:C.muted, marginTop:6 }}>Wrong: {ctxScore.wrong} · 화이팅!</div>
        <div style={{ display:"flex", gap:8, justifyContent:"center", marginTop:16 }}>
          <button onClick={startCtx} style={{ padding:"10px 20px", background:C.blue, color:"#fff", border:"none", borderRadius:10, cursor:"pointer", fontWeight:700 }}>Again</button>
          <button onClick={()=>setMode(null)} style={{ padding:"10px 20px", background:C.light, color:C.textSub, border:"none", borderRadius:10, cursor:"pointer", fontWeight:600 }}>← Back</button>
        </div>
      </div>
    );
    if (!currentCtx) return null;
    // Build multiple choice options
    const correctOpt = currentCtx.answer;
    const allOpts = shuf([correctOpt, ...CONTEXT_CARDS.filter(c=>c.answer!==correctOpt).map(c=>c.answer).sort(()=>Math.random()-0.5).slice(0,3)]);
    return (
      <div>
        <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginBottom:12 }}>
          <div style={{ fontSize:12, color:C.muted }}>{ctxIdx+1} / {ctxItems.length}</div>
          <div style={{ fontSize:12, fontWeight:700, color:systemColor(currentCtx.system) }}>{systemLabel(currentCtx.system)}</div>
          <button onClick={()=>setMode(null)} style={{ fontSize:11, padding:"3px 10px", background:"none", border:`1px solid ${C.border}`, borderRadius:8, cursor:"pointer", color:C.muted }}>← Back</button>
        </div>
        <div style={{ background:C.card, borderRadius:16, padding:"28px 20px", textAlign:"center", border:`1.5px solid ${C.border}`, marginBottom:14 }}>
          <div style={{ fontSize:12, color:C.muted, marginBottom:8, textTransform:"uppercase", letterSpacing:1 }}>How do you say…</div>
          <div style={{ fontSize:32, fontWeight:800, color:C.textSub, marginBottom:6 }}>{currentCtx.prompt}</div>
          {ctxChosen && (
            <div style={{ fontSize:12, color:C.muted, fontStyle:"italic", marginTop:4 }}>💡 {currentCtx.hint}</div>
          )}
        </div>
        <div style={{ display:"flex", flexDirection:"column", gap:8, marginBottom:12 }}>
          {allOpts.map((opt,i)=>{
            const chosen = ctxChosen === opt;
            const isCorrect = opt === correctOpt;
            const showResult = ctxChosen !== null;
            return (
              <button key={i} onClick={()=>{
                if(ctxChosen) return;
                setCtxChosen(opt);
                const correct = opt === correctOpt;
                setCtxScore(s=>({right:s.right+(correct?1:0), wrong:s.wrong+(correct?0:1)}));
              }} style={{
                padding:"12px 16px", borderRadius:12, textAlign:"left", fontSize:15, fontWeight:600, cursor: ctxChosen?"default":"pointer",
                border: `1.5px solid ${showResult?(isCorrect?"#1E8449":chosen?"#B83232":C.border):C.border}`,
                background: C.card,
                color: showResult?(isCorrect?"#1E8449":chosen?"#B83232":C.muted):C.text,
              }}>{opt}</button>
            );
          })}
        </div>
        {ctxChosen && (
          <div style={{ background: ctxChosen===correctOpt?C.greenLight:C.redLight, borderRadius:10, padding:"10px 14px", fontSize:13, marginBottom:12, color: ctxChosen===correctOpt?C.green:C.red }}>
            {ctxChosen===correctOpt?"✓ Correct!" : `✕ The answer is: ${correctOpt}`}
          </div>
        )}
        {ctxChosen && (
          <button onClick={()=>{
            const next = ctxIdx+1;
            if(next>=ctxItems.length){setCtxDone(true);}else{setCtxIdx(next);setCtxChosen(null);}
          }} style={{ padding:"11px 28px", background:C.blue, color:"#fff", border:"none", borderRadius:10, cursor:"pointer", fontWeight:700, fontSize:14 }}>
            {ctxIdx+1>=ctxItems.length?"Finish":"Next →"}
          </button>
        )}
      </div>
    );
  }

  // ── FILL IN THE BLANK MODE ────────────────────────────────────────────────
  if (mode === 'fill') {
    const fill = FILL_BLANKS[fillIdx];
    return (
      <div>
        <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginBottom:12 }}>
          <div style={{ fontSize:12, color:C.muted }}>Sentence {fillIdx+1} / {FILL_BLANKS.length}</div>
          <button onClick={()=>setMode(null)} style={{ fontSize:11, padding:"3px 10px", background:"none", border:`1px solid ${C.border}`, borderRadius:8, cursor:"pointer", color:C.muted }}>← Back</button>
        </div>
        <div style={{ background:C.card, borderRadius:16, padding:"24px 18px", border:`1.5px solid ${C.border}`, marginBottom:14 }}>
          <div style={{ fontSize:12, color:C.muted, marginBottom:10, textTransform:"uppercase", letterSpacing:1 }}>Fill in the blanks</div>
          {/* Sentence display */}
          <div style={{ fontSize:18, fontWeight:700, color:C.textSub, lineHeight:2.2, flexWrap:"wrap", display:"flex", alignItems:"center", gap:4 }}>
            {fill.template.map((part, pi) => {
              const blankIdx = pi > 0 ? pi - 1 : -1;
              const hasBlank = pi < fill.template.length - 1;
              const blankI = pi; // each inter-segment is a blank
              const myBlank = pi < fill.blanks.length ? fill.blanks[pi] : null;
              return (
                <span key={pi} style={{ display:"flex", alignItems:"center", gap:4, flexWrap:"wrap" }}>
                  <span>{part}</span>
                  {pi < fill.blanks.length && (() => {
                    const b = fill.blanks[pi];
                    const chosen = fillAnswers[pi];
                    const isRight = fillChecked && chosen === b.answer;
                    const isWrong = fillChecked && chosen && chosen !== b.answer;
                    return (
                      <span style={{
                        display:"inline-block", minWidth:60, padding:"2px 10px", borderRadius:8,
                        border:`2px solid ${isRight?"#1E8449":isWrong?"#B83232":chosen?C.blue:C.border}`,
                        background: isRight?C.greenLight:isWrong?C.redLight:chosen?C.blueLight:"#f8f8f8",
                        color: isRight?"#1E8449":isWrong?"#B83232":chosen?C.blue:C.muted,
                        fontSize:16, fontWeight:700, textAlign:"center"
                      }}>{chosen || "___"}</span>
                    );
                  })()}
                </span>
              );
            })}
          </div>
          <div style={{ fontSize:13, color:C.muted, marginTop:8, fontStyle:"italic" }}>{fill.en}</div>
        </div>

        {/* Blank selectors */}
        {fill.blanks.map((b,bi)=>(
          <div key={bi} style={{ marginBottom:12 }}>
            <div style={{ fontSize:12, fontWeight:600, color:C.muted, marginBottom:6 }}>Blank {bi+1}: {b.label} {fillChecked && <span style={{ color: fillAnswers[bi]===b.answer?"#1E8449":"#B83232" }}>{fillAnswers[bi]===b.answer?"✓":"✕ "+b.answer}</span>}</div>
            <div style={{ display:"flex", gap:8, flexWrap:"wrap" }}>
              {b.opts.map(opt=>(
                <button key={opt} onClick={()=>handleFillPick(bi,opt)} style={{
                  padding:"8px 14px", borderRadius:10, fontSize:14, fontWeight:700, cursor:fillChecked?"default":"pointer",
                  border:`1.5px solid ${fillAnswers[bi]===opt?(fillChecked?(opt===b.answer?"#1E8449":"#B83232"):C.blue):C.border}`,
                  background: fillAnswers[bi]===opt?(fillChecked?(opt===b.answer?C.greenLight:C.redLight):C.blueLight):C.card,
                  color: fillAnswers[bi]===opt?(fillChecked?(opt===b.answer?"#1E8449":"#B83232"):C.blue):C.text,
                }}>{opt}</button>
              ))}
            </div>
            {fillChecked && <div style={{ fontSize:12, color:C.muted, marginTop:4, fontStyle:"italic" }}>💡 {b.hint}</div>}
          </div>
        ))}

        {fillChecked && (
          <div style={{ background: fillCorrect?C.greenLight:C.redLight, borderRadius:10, padding:"10px 14px", fontSize:13, marginBottom:12, color:fillCorrect?C.green:C.red }}>
            {fillCorrect?"✓ Perfect!" : "✕ See corrections above"}
            <div style={{ fontSize:12, color:C.muted, marginTop:4 }}>{fill.note}</div>
          </div>
        )}

        <div style={{ display:"flex", gap:8 }}>
          {!fillChecked ? (
            <button onClick={()=>setFillChecked(true)} disabled={!allFillAnswered} style={{ padding:"11px 24px", background:allFillAnswered?C.blue:"#ccc", color:"#fff", border:"none", borderRadius:10, cursor:allFillAnswered?"pointer":"default", fontWeight:700, fontSize:14 }}>Check ✓</button>
          ) : (
            <button onClick={()=>{
              const next=fillIdx+1;
              if(next>=FILL_BLANKS.length){setFillIdx(0);}else{setFillIdx(next);}
              setFillAnswers([]); setFillChecked(false);
